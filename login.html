from pyodide import create_proxy from js import console, fetch, XMLHttpRequest, Blob, window import json loginBtn = document.getElementById('login') def login(event): email = document.getElementById('email').value password = document.getElementById('password').value console.log("login") console.log(email) console.log(password) data = { "email": email, "password": password } console.log(data) req = XMLHttpRequest.new() req.open("POST", "http://127.0.0.1:8000/api/login/", False) req.setRequestHeader("Content-Type", "application/json;charset=UTF-8") blob = Blob.new([json.dumps(data)], {type : 'application/json'}) console.log(blob) req.send(blob) a = (req.response) if req.status == 200: window.localStorage.setItem('token', json.loads(a).get('jwt')) window.localStorage.setItem('role', json.loads(a).get('role')) window.localStorage.setItem('id', json.loads(a).get('id')) if json.loads(a).get('role') == 1 or json.loads(a).get('role') == 2: window.location.href = "/user.html" else: window.location.href = "/member.html" loginEvent = create_proxy(login) loginBtn.addEventListener('click', loginEvent)
